% custom small space title
\newcommand{\smallspacechapter}[2][]{
  {
    \titleformat{\chapter}[display]
        {\normalfont\huge\bfseries} % Custom formatting
        {}                          % Label
        {0pt}                       % Space between label and title
        {\Huge}                     % Title text style

    \titlespacing*{\chapter}{0pt}{-20pt}{0pt}

    \ifthenelse{\equal{#1}{}}%
      {\chapter{#2}}% If no short title is provided, use only the full title
      {\chapter[#1]{#2}}% If a short title is provided, use it for TOC
  }
}
